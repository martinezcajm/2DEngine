258 struct IMGUI_API ImRect
259 {
260  ImVec2 Min; // Upper-left
261  ImVec2 Max; // Lower-right
262 
263  ImRect() : Min(FLT_MAX,FLT_MAX), Max(-FLT_MAX,-FLT_MAX) {}
264  ImRect(const ImVec2& min, const ImVec2& max) : Min(min), Max(max) {}
265  ImRect(const ImVec4& v) : Min(v.x, v.y), Max(v.z, v.w) {}
266  ImRect(float x1, float y1, float x2, float y2) : Min(x1, y1), Max(x2, y2) {}
267 
268  ImVec2 GetCenter() const { return ImVec2((Min.x+Max.x)*0.5f, (Min.y+Max.y)*0.5f); }
269  ImVec2 GetSize() const { return ImVec2(Max.x-Min.x, Max.y-Min.y); }
270  float GetWidth() const { return Max.x-Min.x; }
271  float GetHeight() const { return Max.y-Min.y; }
272  ImVec2 GetTL() const { return Min; } // Top-left
273  ImVec2 GetTR() const { return ImVec2(Max.x, Min.y); } // Top-right
274  ImVec2 GetBL() const { return ImVec2(Min.x, Max.y); } // Bottom-left
275  ImVec2 GetBR() const { return Max; } // Bottom-right
276  bool Contains(const ImVec2& p) const { return p.x >= Min.x && p.y >= Min.y && p.x < Max.x && p.y < Max.y; }
277  bool Contains(const ImRect& r) const { return r.Min.x >= Min.x && r.Min.y >= Min.y && r.Max.x < Max.x && r.Max.y < Max.y; }
278  bool Overlaps(const ImRect& r) const { return r.Min.y < Max.y && r.Max.y > Min.y && r.Min.x < Max.x && r.Max.x > Min.x; }
279  void Add(const ImVec2& rhs) { if (Min.x > rhs.x) Min.x = rhs.x; if (Min.y > rhs.y) Min.y = rhs.y; if (Max.x < rhs.x) Max.x = rhs.x; if (Max.y < rhs.y) Max.y = rhs.y; }
280  void Add(const ImRect& rhs) { if (Min.x > rhs.Min.x) Min.x = rhs.Min.x; if (Min.y > rhs.Min.y) Min.y = rhs.Min.y; if (Max.x < rhs.Max.x) Max.x = rhs.Max.x; if (Max.y < rhs.Max.y) Max.y = rhs.Max.y; }
281  void Expand(const float amount) { Min.x -= amount; Min.y -= amount; Max.x += amount; Max.y += amount; }
282  void Expand(const ImVec2& amount) { Min.x -= amount.x; Min.y -= amount.y; Max.x += amount.x; Max.y += amount.y; }
283  void Translate(const ImVec2& v) { Min.x += v.x; Min.y += v.y; Max.x += v.x; Max.y += v.y; }
284  void ClipWith(const ImRect& clip) { if (Min.x < clip.Min.x) Min.x = clip.Min.x; if (Min.y < clip.Min.y) Min.y = clip.Min.y; if (Max.x > clip.Max.x) Max.x = clip.Max.x; if (Max.y > clip.Max.y) Max.y = clip.Max.y; }
285  void Floor() { Min.x = (float)(int)Min.x; Min.y = (float)(int)Min.y; Max.x = (float)(int)Max.x; Max.y = (float)(int)Max.y; }
286  ImVec2 GetClosestPoint(ImVec2 p, bool on_edge) const
287  {
288  if (!on_edge && Contains(p))
289  return p;
290  if (p.x > Max.x) p.x = Max.x;
291  else if (p.x < Min.x) p.x = Min.x;
292  if (p.y > Max.y) p.y = Max.y;
293  else if (p.y < Min.y) p.y = Min.y;
294  return p;
295  }
